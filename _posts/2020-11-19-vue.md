### 计算属性
```
<body>
    <div id="cc">
        <p>{{message}}</p>
        <p>{{reversedMessage}}</p>
    </div>
</body>

<script>
    new Vue({
        el:'#cc',                   //el是挂载点
        data:{
            message:'hello'
        },
        computed:{
            reversedMessage:function (){
                return this.message.split('').reverse().join('')
            }
        }
    })
```
